Nginx Error

filter {
  if [format] == "string" and [type] == "error" and [package] == "nginx" {
    grok {
      match => [ "message", "(?<timestamp>[0-9A-z\/:\ ]{0,19}) \[%{LOGLEVEL:level}\] %{GREEDYDATA:payload}" ]
    }
  }
}
